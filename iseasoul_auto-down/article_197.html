B 可以证明 N - 2 即为答案
N - 2为上限，显然
N 不可能 N - 1时只有一个点的度最小，也不会出现
N - 2的构造：
完全图 任选两点 去掉之间的边
因此 这两点的度为 N - 2 其他点的度为N - 1
而去掉这两个点后 其他点的度变为 N - 3 安全





C 按a排序 对每两个，选取方法：
如果p(i).b < p(i + 1).b 选取后一个 这个很明显
否则 选取前一个
证明：对第二种选取方式 sum(a) 和 sum(b) 都比未选取部分的和大
sum(a) 有p(2 * n - 1).a >= p(2 * n - 2).a ... p(3).a >= p(2).a 因此加上p(1).a后，一定大于
sum(b) 每次选取都保证p(i).b >= p(i + 1).b 加上p(2 * n - 1).b后，一定大于

#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <queue>
#include <vector>
#include <map>

using namespace std;

const int maxn = 200000 + 10;
const int inf = (-1u) >> 1;

struct item {
    int a, b, d;
    bool operator<(const item& t) const {
        return a < t.a; } };

item p[maxn];

int main() {
    int t, n;
    scanf ("%d", &t);
    while (t--) {
        scanf ("%d", &n);
        n = (n << 1) - 1;
        for (int i = 0; i < n; ++i) {
            scanf ("%d%d", &p[i].a, &p[i].b);
            p[i].d = i + 1; }
        sort (p, p + n);
        puts ("YES");
        for (int i = 0; i < n - 1; i += 2) 
            printf ("%d ", p[p[i].b < p[i + 1].b? i + 1 : i].d);
        printf ("%d\n", p[n - 1].d); }
    return 0; }
