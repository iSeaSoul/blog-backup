bool dfs(int x)
{
if (v[x])
return false;
v[x] = true;
for (int i = 0; i < n; ++i) {
if (g[x][i] && (rMatch[i] == -1 || dfs(rMatch[i]))) {
lMatch[x] = i;
rMatch[i] = x;
return true;
}
}
return false;
}












int maxMatch()
{
int res = 0;
memset (lMatch, -1, sizeof(lMatch));
memset (rMatch, -1, sizeof(rMatch));
bool flag;
do {
memset (v, 0, sizeof(v));
flag = false;
for (int i = 0; i < m; ++i) {
if (lMatch[i] == -1 && dfs(i)) {
flag = true;
++res;
}
}
}while (flag);
return res;
}
