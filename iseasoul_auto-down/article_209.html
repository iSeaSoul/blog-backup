A题：一句话的力量有多大？for (int j = state; ; j = (j - 1) & state)

#include <cstdio>
#include <algorithm>

using namespace std;

const int maxn = 50000 + 10;
const int maxm = (1 << 16) + 10;

int a[maxn], n, m, dp[maxm];

void init()
{
for (int i = 0, t; i < m; ++i) {
scanf ("%d", &t);
a[i] = 0;
for (int j = 0, p; j < t; ++j) {
scanf ("%d", &p);
a[i] |= (1 << (p - 1));
}
}
}









void solve()
{
int target = (1 << n) - 1;
fill (dp, dp + (1 << n), 0);
for (int i = 0; i < m; ++i) {
int state = (target ^ a[i]);
for (int j = state; ; j = (j - 1) & state) {
dp[a[i] | j] = max(dp[a[i] | j], dp[j] + 1);
if (j == 0)
break;
}
}
printf ("%d\n", dp[target]);
}












int main()
{
while (scanf ("%d%d", &n, &m) != EOF) {
init();
solve();
}

return 0;
}

B题：为什么这么像图论... 其实还是暴力...

#include <cstdio>
#include <algorithm>

using namespace std;

const int goal = (1 << 20);

char m[22][22];
int n, state[goal], col[22], t[22];

void premake()
{
for (int i = 0; i < goal; ++i) {
state[i] = 0;
for (int j = 0; j < 20; ++j)
if ((i >> j) & 1)
++state[i];
}
}







void init()
{
for (int i = 0; i < 20; ++i)
scanf ("%s", m[i]);
for (int j = 0; j < 20; ++j) {
col[j] = 0;
for (int i = 0; i < 20; ++i)
if (m[i][j] == '#')
col[j] |= (1 << i);
}
}









void solve()
{
int ans = 0;
for (int i = 0; i < goal; ++i) {
if (state[i] < n) {
for (int j = 0; j < 20; ++j)
t[j] = state[col[j] & i];
sort (t, t + 20);
int now = 0;
for (int k = max(0, 20 - (n - state[i])); k < 20; ++k)
now += t[k];
ans = max(ans, now);
}
}
printf ("%d\n", ans);
}














int main()
{
premake();
while (scanf ("%d", &n), n) {
init();
solve();
}

return 0;
}

C题：要是没有前两题，这道题特定被板刷...

#include <cstdio>
#include <algorithm>

using namespace std;

const int maxn = 200 + 10;

int c[maxn], d[maxn], n, k;
double a, b, dp[maxn][maxn], res[maxn];

void cal(int l, int r)
{
double per = dp[l][r];
int num = r - l + 1;
for (int i = l; i <= r; ++i) {
res[i] += per * a / num;
if (i - 1 >= 0)
dp[l][i - 1] += per * b / num;
if (i + 1 < n)
dp[i + 1][r] += per * (1 - a - b) / num;
}
}










void init()
{
for (int i = 0; i < n; ++i)
scanf ("%d", &c[i]);
scanf ("%d%lf%lf", &k, &a, &b);
}




void solve()
{
fill (res, res + n, 0.0);
for (int i = 0; i < n; ++i)
fill (dp[i], dp[i] + n, 0.0);
dp[0][n - 1] = 1.0;
for (int len = n - 1; len >= 0; --len) {
for (int j = 0; j + len < n; ++j)
cal(j, j + len);
}
copy (c, c + n, d);
sort (d, d + n);
int newn = unique(d, d + n) - d;
if (k > newn) {
printf ("%.3lf\n", 0.0);
return ;
}
double ans = 0.0;
for (int i = 0; i < n; ++i)
if (c[i] == d[k - 1])
ans += res[i];
printf ("%.3lf\n", ans);
}





















int main()
{
while (scanf ("%d", &n), n > 0) {
init();
solve();
}

return 0;
}
