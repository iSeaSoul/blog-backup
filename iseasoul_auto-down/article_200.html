模线性方程 暨 POJ 1061

22th of June 2010

默认分类

形如ax % b = c 设 -y = ax / b则方程亦为ax + by = c
这个方程假如有解，则x + b / gcd(a, b), y C a / gcd(a, b) 也为其解，并且这个是最小的解步长
利用扩展欧几里得求出的x和y实际上是ax + by = gcd(a, b)的一组解
所以模线性方程有解的必要条件是c | gcd(a, b)


扩展欧几里得的代码： 

int exgcd(int a, int b, int& x, int& y) {
     if (b == 0) {
         x = 1, y = 0;
         return a; }
     int r = exgcd(b, a % b, x, y);
     int t = x;
     x = y;
     y = t - a / b * y;
     return r; }

解释：求出 exgcd(b, a % b, x, y) 后，满足b * x + a % b * y = gcd(b, a % b)
而我们需要a * x + b * y = gcd(a, b)
代换 即gcd(a, b) = b * x + a % b * y = b * x + (a C [a / b] * b) * y = a * y + b * (x C [a / b] * y)
则有x_new = y, y_new = x C a / b * y


扩展欧几里得求出x与y后，乘以c / gcd(a, b)即可

1061，基础的模线性方程 代码： 

/*

 * Created Time:  2010/6/22 13:48:21
 * File Name: POJ_1061.cpp
 */
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <queue>
#include <vector>
#include <map>

using namespace std;

const int maxn = 1000 + 10;
const int inf = (-1u) >> 1;

typedef long long int64;

int64 x, y, m, n, l;

int64 exgcd(int64 a, int64 b, int64& x, int64& y) {
    if (b == 0) {
        x = 1, y = 0;
        return a; }
    int64 r = exgcd(b, a % b, x, y);
    int64 t = x;
    x = y;
    y = t - a / b * y;
    return r; }

void solve() {
    int64 a = m >= n? m - n : l - n + m;
    int64 b = y >= x? y - x : l - x + y, c, d;
    int64 gal = exgcd(a, l, c, d);
    if (b % gal != 0) {
        puts ("Impossible");
        return ; }
    c *= b / gal;
    int64 stepc = l / gal;
    c %= stepc;
    printf ("%I64d\n", c < 0? (stepc + c) % stepc : c); }

int main() {
    while (scanf ("%I64d%I64d%I64d%I64d%I64d", &x, &y, &m, &n, &l) != EOF)
        solve();
    return 0; }
