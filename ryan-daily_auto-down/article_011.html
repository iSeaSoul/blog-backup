Today, let's begin with a useful, interesting and tricky topic: Reference & Pointer.

Reference is an alias of a variable.

Pointer can be treated as an address, and its type indicate the length of the data it points. 

namespace test_reference_and_pointer {

void test() {
    int x = -1;
    int& r_x = x, r_y = x; // int& x <=> int &x
    r_x = 0;
    cout << x << ' ' << r_x << ' ' << r_y << endl;
    r_y = 10;
    cout << x << ' ' << r_x << ' ' << r_y << endl;

    // How to define a pointer by a ref or a variable
    int *p_x = &r_x;
    *p_x = 10;
    cout << x << ' ' << r_x << ' ' << *p_x << endl;
    p_x = &x;
    *p_x = 12;
    cout << x << ' ' << r_x << ' ' << *p_x << endl;

    // How to define a ref by a pointer
    int &r_p_x = *p_x;
    r_p_x = 15;
    cout << x << ' ' << r_x << ' ' << *p_x << endl;

    // the meaning of a pointer to be nullptr
    int* p_x_2 = &x;
    x = 0;
    if (p_x_2) {
        cout << "p_x_2 not empty" << endl;
    } else {
        cout << "p_x_2 empty" << endl;
    }
    p_x_2 = NULL;
    if (p_x_2) {
        cout << "p_x_2 not empty" << endl;
    } else {
        cout << "p_x_2 empty" << endl;
    }

    // the ref of a pointer
    int *&r_p_x_2 = p_x_2;
    r_p_x_2 = &x;
    if (p_x_2) {
        cout << "p_x_2 not empty" << endl;
    } else {
        cout << "p_x_2 empty" << endl;
    }
}

}
