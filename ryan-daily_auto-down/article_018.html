source $VIMRUNTIME/mswin.vim
behave mswin
cd d:/prog
imap <c-d> <c-o>dd
imap <cr> <cr><left><right>
map o o<left><right>
imap <c-]> {<cr>}<c-o>O<left><right>
noremap <f6> =a{
syn on
colo desertEx
filetype indent on
se ru nu ar sw=4 ts=4 noswf et sta nowrap ww=<,>,[,] gfn=Consolas:h14
autocmd BufEnter * lcd %:p:h

map vs :vsplit
map <c-t> :tabnew<CR>
map <c-w> :q<CR>

"set encoding=utf-8

map <c-b> :call CR()<CR><CR>
func CR()
exec 'update'
if filereadable(expand('%<').'.in')
    exec '!start cmd /c g++ -std=c++11 %<.cpp -o %<.exe & %< < %<.in & pause'
else
    exec '!start cmd /c g++ -std=c++11 %<.cpp -o %<.exe & %< & pause'
endif
endfunc

map <f11> :call PY()<CR><CR>
func PY()
exec 'update'
if filereadable(expand('%<').'.in')
    exec '!start cmd /c python %<.py < %<.in & pause'
else
    exec '!start cmd /c python %<.py & pause'
endif
endfunc

map <f12> :call JA()<CR><CR>
func JA()
exec 'update'
if filereadable(expand('%<').'.in')
    exec '!start cmd /c javac %<.java & java %< < %<.in & pause'
else
    exec '!start cmd /c javac %<.java & java %< & pause'
endif
endfunc

map<f4> :call AddComment()<CR>
func AddComment()
    if matchstr(getline('.'), '[^ ]') == '/'
        normal ^xx
    else
        normal ^i//
    endif
endfunc
